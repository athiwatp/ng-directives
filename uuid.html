UUID Service

A very simple service for working with UUIDs

{{new}}
{{nInfo | date:'mediumDate'}} {{nInfo | date:'mediumTime'}}
{{empty}}
{{eInfo | date:'mediumDate'}} {{eInfo | date:'mediumTime'}}

Page script

		angular
			.module('uuidApp', ['lvl.services'])
			.controller('uuidCtl', ['$scope', 'uuid', function($scope, uuid){
				$scope.generateUuid = function() {
					$scope.new = uuid.new();
					$scope.nInfo = new Date();
				};

				$scope.showEmpty = function() {
					$scope.empty = uuid.empty();
					$scope.eInfo = new Date();
				};
			}]);
